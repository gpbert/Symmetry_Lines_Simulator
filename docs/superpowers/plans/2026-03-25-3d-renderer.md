```javascript
function initScene() {
    scene = new THREE.Scene();
    scene.background = new THREE.Color(0xf5f5f5);

    // Camera
    camera = new THREE.PerspectiveCamera(45, 1, 0.1, 1000);
    camera.position.set(30, 40, 30);
    camera.lookAt(0, 0, 0);

    // Renderer
    renderer = new THREE.WebGLRenderer({ antialias: true });
    renderer.setPixelRatio(window.devicePixelRatio);
    renderer.shadowMap.enabled = true;
    renderer.shadowMap.type = THREE.PCFSoftShadowMap;

    // Lights
    const ambient = new THREE.AmbientLight(0x808080);
    scene.add(ambient);
    const dirLight = new THREE.DirectionalLight(0xffffff, 1.2);
    dirLight.position.set(40, 80, 40);
    dirLight.castShadow = true;
    dirLight.shadow.mapSize.set(2048, 2048);
    const d = 100;
    dirLight.shadow.camera.left = -d;
    dirLight.shadow.camera.right = d;
    dirLight.shadow.camera.top = d;
    dirLight.shadow.camera.bottom = -d;
    scene.add(dirLight);

    // Ground plane for raycasting
    const planeGeo = new THREE.PlaneGeometry(500, 500);
    const planeMat = new THREE.MeshBasicMaterial({ visible: false });
    groundPlane = new THREE.Mesh(planeGeo, planeMat);
    groundPlane.rotation.x = -Math.PI / 2;
    scene.add(groundPlane);

    // Groups
    gridGroup = new THREE.Group();
    envelopeGroup = new THREE.Group();
    zoneGroup = new THREE.Group();
    previewGroup = new THREE.Group();
    scene.add(gridGroup);
    scene.add(envelopeGroup);
    scene.add(zoneGroup);
    scene.add(previewGroup);
}
```

```javascript
function buildGrid() {
    // Clear existing
    while (gridGroup.children.length) gridGroup.remove(gridGroup.children[0]);

    const size = 200; // units (20,000mm = 20m each side)

    // 100mm internal grid (thin, light)
    const internalDiv = Math.floor(size / mmToUnits(GRID_SIZE_INTERNAL));
    const internalGrid = new THREE.GridHelper(size, internalDiv, 0xdddddd, 0xdddddd);
    internalGrid.position.y = 0.001;
    gridGroup.add(internalGrid);

    // 300mm external grid (thick, dark)
    const externalDiv = Math.floor(size / mmToUnits(GRID_SIZE_EXTERNAL));
    const externalGrid = new THREE.GridHelper(size, externalDiv, 0x999999, 0x999999);
    externalGrid.position.y = 0.002;
    gridGroup.add(externalGrid);
}
```

```javascript
// Materials
const wallMat = new THREE.MeshStandardMaterial({ color: 0xcccccc, roughness: 0.6 });
const wallInternalFaceMat = new THREE.MeshStandardMaterial({ color: 0x4488cc, roughness: 0.5 }); // Blue for column face
const wallSelectedMat = new THREE.MeshStandardMaterial({ color: 0x2563eb, roughness: 0.5 });
const wallGhostMat = new THREE.MeshStandardMaterial({ color: 0x888888, transparent: true, opacity: 0.2, roughness: 0.5 });
const columnMat = new THREE.MeshStandardMaterial({ color: 0x555555, roughness: 0.3 });

function createWallMesh(wall, floorY) {
    const length = mmToUnits(wall.length);
    const height = mmToUnits(wall.height);
    const thickness = mmToUnits(wall.thickness);

    // Wall body
    const geo = new THREE.BoxGeometry(thickness, height, length);
    const mesh = new THREE.Mesh(geo, wallMat);

    // Position at wall midpoint
    const midX = mmToUnits((wall.pointA.x + wall.pointB.x) / 2);
    const midZ = mmToUnits((wall.pointA.y + wall.pointB.y) / 2);
    // Offset by half thickness in normal direction (wall geometry centered, but A/B are on internal face)
    const offsetX = mmToUnits(wall.n.x * wall.thickness / 2);
    const offsetZ = mmToUnits(wall.n.y * wall.thickness / 2);

    mesh.position.set(midX + offsetX, floorY + height / 2, midZ + offsetZ);

    // Rotate to match wall direction
    const angle = Math.atan2(wall.d.x, wall.d.y);
    mesh.rotation.y = angle;

    mesh.castShadow = true;
    mesh.receiveShadow = true;

    // Steel columns at endpoints
    const colSize = mmToUnits(COLUMN_SIZE);
    const colHeight = height;
    const colGeo = new THREE.BoxGeometry(colSize, colHeight, colSize);

    const columnMeshes = [];
    [wall.pointA, wall.pointB].forEach(pt => {
        const colMesh = new THREE.Mesh(colGeo, columnMat);
        // Column on internal face, centered on the column
        const cx = mmToUnits(pt.x) + mmToUnits(wall.n.x * COLUMN_SIZE / 2);
        const cz = mmToUnits(pt.y) + mmToUnits(wall.n.y * COLUMN_SIZE / 2);
        colMesh.position.set(cx, floorY + colHeight / 2, cz);
        colMesh.castShadow = true;
        columnMeshes.push(colMesh);
    });

    return { mesh, columnMeshes };
}
```

- [ ] **Step 3: Add full draw function**

```javascript
function draw() {
    if (!scene) return;

    // Clear dynamic objects
    wallMeshes.forEach(wm => {
        scene.remove(wm.mesh);
        wm.columnMeshes.forEach(cm => scene.remove(cm));
    });
    wallMeshes = [];
    voidMeshes.forEach(vm => scene.remove(vm.mesh));
    voidMeshes = [];
    while (envelopeGroup.children.length) envelopeGroup.remove(envelopeGroup.children[0]);
    while (zoneGroup.children.length) zoneGroup.remove(zoneGroup.children[0]);
    while (previewGroup.children.length) previewGroup.remove(previewGroup.children[0]);

    // Update grid Y position for current floor
    const currentFloorY = state.currentFloorId * FLOOR_HEIGHT_UNITS;
    gridGroup.position.y = currentFloorY;
    groundPlane.position.y = currentFloorY;

    // Render walls
    state.walls.forEach(wall => {
        const floorY = wall.floorId * FLOOR_HEIGHT_UNITS;
        const isCurrentFloor = wall.floorId === state.currentFloorId;
        const isBelow = wall.floorId < state.currentFloorId;
        const isAbove = wall.floorId > state.currentFloorId;

        // Section cut: hide floors above
        if (isAbove) return;
        // Show below only if toggle is on
        if (isBelow && !state.showLevelsBelow) return;

        const { mesh, columnMeshes } = createWallMesh(wall, floorY);

        if (!isCurrentFloor) {
            mesh.material = wallGhostMat;
            columnMeshes.forEach(cm => { cm.material = wallGhostMat; });
        } else if (state.selectedWalls.includes(wall)) {
            mesh.material = wallSelectedMat;
        }

        scene.add(mesh);
        columnMeshes.forEach(cm => scene.add(cm));
        wallMeshes.push({ wall, mesh, columnMeshes });
    });

    // Render voids
    state.voids.forEach(v => {
        const floorY = v.floorId * FLOOR_HEIGHT_UNITS;
        const isCurrentFloor = v.floorId === state.currentFloorId;
        const isBelow = v.floorId < state.currentFloorId;
        const isAbove = v.floorId > state.currentFloorId;

        if (isAbove) return;
        if (isBelow && !state.showLevelsBelow) return;

        const mesh = createVoidMesh(v, floorY);
        if (!isCurrentFloor) mesh.material = voidGhostMat;
        scene.add(mesh);
        voidMeshes.push({ void: v, mesh });
    });

    // Render building envelopes as floor planes
    state.buildingEnvelopes.forEach(env => {
        if (env.floorId !== state.currentFloorId) return;
        if (!env.polygon || env.polygon.length < 3) return;

        const shape = new THREE.Shape();
        const p0 = env.polygon[0];
        shape.moveTo(mmToUnits(p0.x), mmToUnits(p0.y));
        for (let i = 1; i < env.polygon.length; i++) {
            shape.lineTo(mmToUnits(env.polygon[i].x), mmToUnits(env.polygon[i].y));
        }
        shape.lineTo(mmToUnits(p0.x), mmToUnits(p0.y));

        const geo = new THREE.ShapeGeometry(shape);
        const mat = new THREE.MeshBasicMaterial({
            color: 0x2563eb, transparent: true, opacity: 0.05, side: THREE.DoubleSide
        });
        const mesh = new THREE.Mesh(geo, mat);
        mesh.rotation.x = -Math.PI / 2;
        mesh.position.y = currentFloorY + 0.005;
        envelopeGroup.add(mesh);
    });

    // Render preview wall
    const istate = _interactionState;
    if (istate.drawingWall && istate.tempPoint && state.currentMode === 'draw') {
        const start = istate.drawingWall;
        const end = istate.tempPoint;
        const dx = end.x - start.x;
        const dy = end.y - start.y;
        const length = Math.sqrt(dx * dx + dy * dy);
        if (length > 0) {
            const previewGeo = new THREE.BoxGeometry(
                mmToUnits(200), // default thickness preview
                FLOOR_HEIGHT_UNITS,
                mmToUnits(length)
            );
            const previewMat = new THREE.MeshStandardMaterial({
                color: 0x22c55e, transparent: true, opacity: 0.4
            });
            const previewMesh = new THREE.Mesh(previewGeo, previewMat);
            const midX = mmToUnits((start.x + end.x) / 2);
            const midZ = mmToUnits((start.y + end.y) / 2);
            previewMesh.position.set(midX, currentFloorY + FLOOR_HEIGHT_UNITS / 2, midZ);
            const angle = Math.atan2(dx, dy);
            previewMesh.rotation.y = angle;
            previewGroup.add(previewMesh);
        }
    }

    // Render void preview
    if (istate.drawingVoid && istate.tempPoint && state.currentMode === 'void') {
        const startX = Math.min(istate.drawingVoid.startX, istate.tempPoint.x);
        const startY = Math.min(istate.drawingVoid.startY, istate.tempPoint.y);
        const w = Math.abs(istate.tempPoint.x - istate.drawingVoid.startX);
        const h = Math.abs(istate.tempPoint.y - istate.drawingVoid.startY);
        if (w > 0 && h > 0) {
            const geo = new THREE.PlaneGeometry(mmToUnits(w), mmToUnits(h));
            const tooSmall = w < MIN_VOID_SIZE || h < MIN_VOID_SIZE;
            const mat = new THREE.MeshBasicMaterial({
                color: tooSmall ? 0xdc2626 : 0x22c55e,
                transparent: true, opacity: 0.3, side: THREE.DoubleSide
            });
            const mesh = new THREE.Mesh(geo, mat);
            mesh.rotation.x = -Math.PI / 2;
            mesh.position.set(
                mmToUnits(startX + w / 2),
                currentFloorY + 0.02,
                mmToUnits(startY + h / 2)
            );
            previewGroup.add(mesh);
        }
    }
}
```

```javascript
function screenToWorld(event) {
    if (!container || !camera) return { x: 0, y: 0 };
    const rect = container.getBoundingClientRect();
    mouse.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
    mouse.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

    raycaster.setFromCamera(mouse, camera);
    const intersects = raycaster.intersectObject(groundPlane);
    if (intersects.length > 0) {
        const p = intersects[0].point;
        let x = unitsToMm(p.x);
        let y = unitsToMm(p.z); // Three.js Z = sim Y
        x = sim.snapToGrid(x, sim.GRID_SIZE_EXTERNAL);
        y = sim.snapToGrid(y, sim.GRID_SIZE_EXTERNAL);
        return { x, y, screenX: event.clientX - rect.left, screenY: event.clientY - rect.top };
    }
    return null;
}
```

```javascript
export const renderer3D = {
    init(cont) {
        container = cont;
        initScene();
        renderer.setSize(cont.clientWidth, cont.clientHeight);
        cont.appendChild(renderer.domElement);
        buildGrid();

        // OrbitControls
        controls = new OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;
        controls.dampingFactor = 0.05;
        controls.maxPolarAngle = Math.PI / 2 - 0.05;
        controls.mouseButtons = {
            LEFT: null, // We handle left-click for tools
            MIDDLE: THREE.MOUSE.PAN,
            RIGHT: THREE.MOUSE.ROTATE,
        };
        controls.enabled = false; // Disabled until activated
    },

    draw: draw,

    activate() {
        container.style.display = '';
        controls.enabled = true;
        window.addEventListener('resize', this._onResize);
        this._onResize();
        this._startAnimationLoop();
    },

    deactivate() {
        container.style.display = 'none';
        controls.enabled = false;
        window.removeEventListener('resize', this._onResize);
        this._stopAnimationLoop();
    },

    screenToWorld: screenToWorld,

    getCanvas() { return renderer ? renderer.domElement : null; },
    get panOffset() { return { x: 0, y: 0 }; }, // Not used in 3D, but interface compat
    get zoomLevel() { return 1; }, // Not used in 3D
    get isPanning() { return false; },
    set isPanning(val) { /* noop */ },

    setInteractionState(istate) { _interactionState = istate; },

    _onResize() {
        if (!container || !camera || !renderer) return;
        const w = container.clientWidth;
        const h = container.clientHeight;
        camera.aspect = w / h;
        camera.updateProjectionMatrix();
        renderer.setSize(w, h);
    },

    _bindNavigation() { /* OrbitControls handles this */ },
    _unbindNavigation() { /* OrbitControls handles this */ },

    _startAnimationLoop() {
        const animate = () => {
            animationId = requestAnimationFrame(animate);
            controls.update();
            if (renderer && scene && camera) {
                renderer.render(scene, camera);
            }
        };
        animate();
    },

    _stopAnimationLoop() {
        if (animationId) {
            cancelAnimationFrame(animationId);
            animationId = null;
        }
    },
};
```
